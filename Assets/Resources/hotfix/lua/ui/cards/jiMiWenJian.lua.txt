JiMiWenJian = {}
JiMiWenJian.__index = JiMiWenJian

Global.JiMiWenJianDealCardsFlag = false

function JiMiWenJian.CheckUseCondition()
	if Global.Data.RoomInfo.CurrentGameTurnUserName ~= Global.Data.User.UserName then
		Global.Methods.ShowTips2("只能在自己回合中使用")
		return false
	end
	return true
end

function JiMiWenJian.OnTriggerCardAnimationEnd(data, script)
	Global.Protos.Game.SendUseJiMiWenJian()
end

function JieHuo.ChooseTarget()
	return false
end

function JiMiWenJian.Dispose()
	Global.JiMiWenJianDealCardsFlag = false
end

return JiMiWenJian
