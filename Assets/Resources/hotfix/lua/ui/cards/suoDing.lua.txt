SuoDing = {}
SuoDing.__index = SuoDing

function SuoDing.CheckUseCondition()
	if Global.Data.RoomInfo.CurrentGameTurnUserName ~= Global.Data.User.UserName then
		Global.Methods.ShowTips2("只能在自己回合中使用")
		return false
	end
	return true
end

function SuoDing.OnTriggerCardAnimationEnd(data, script)
	Global.Protos.Game.SendUseSuoDing()
end

function SuoDing.ChooseTarget()
	Global.Methods.ShowTips("请选择你要锁定的玩家")
	EventManager:TriggerEvent(Global.Event.Card.SuoDingChoose, nil)
	return true
end

function SuoDing.Dispose()

end

return SuoDing
