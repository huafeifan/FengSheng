ZengYuan = {}
ZengYuan.__index = ZengYuan

Global.ZengYuanDealCardsFlag = false

function ZengYuan.CheckUseCondition()
	if Global.Data.RoomInfo.CurrentGameTurnUserName ~= Global.Data.User.UserName then
		Global.Methods.ShowTips2("只能在自己回合中使用")
		return false
	end
	return true
end

function ZengYuan.OnTriggerCardAnimationEnd(data, script)
	Global.Protos.Game.SendUseZengYuan()
end

function ZengYuan.ChooseTarget()
	return false
end

--在触发效果后,是否立刻结束回合
function ZengYuan.IsEndTurnImmediatelyAfterTrigger()
	return false
end

function ZengYuan.Dispose()
	Global.ZengYuanDealCardsFlag = false
end

return ZengYuan
