UserHead = {}

function UserHead:New(gameObject)
	local item = {}
	item.gameObject = gameObject
	item.transform = gameObject.transform
	item.ui = {}

	setmetatable(item, UserHead)
	return item
end

function UserHead:InitUIReference()
	self.ui.userHead = self.transform:Find("Head"):GetComponent(typeof(Image))
	self.ui.userName = self.transform:Find("NameBg/Name"):GetComponent(typeof(Text))
end
