Global.Protos.Room = {}

Global.Protos.Room.SendRoomList = function()
	NetManager:Send(Global.Login.Network, Global.Protos.Cmd.RoomList_C2S, nil)
end

Global.Protos.Room.SendCreateRoom = function(data)
	local dataBytes = Global.Protos.Methods.Lua2Bytes(Global.Protos.FileName.Room, Global.Protos.Class.CreateRoom_C2S, data)
	NetManager:Send(Global.Login.Network, Global.Protos.Cmd.CreateRoom_C2S, dataBytes)
end

Global.Protos.Room.SendEnterRoom = function(roomNub)
	local sendData = {}
	sendData.userName = Global.Data.User.UserName
	sendData.roomNub = roomNub

	NetManager:Send(Global.Login.Network, Global.Protos.Cmd.EnterRoom_C2S, dataBytes)
end
